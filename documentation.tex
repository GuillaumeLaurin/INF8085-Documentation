\footnotesize \section{Modèle de Shannon}

\begin{center}
    \scriptsize source $\rightarrow$ codeur $\Rightarrow$ décodeur $\rightarrow$ récepteur
\end{center}

\begin{itemize}[leftmargin=0.5em,label=$\bullet$]
    \item \scriptsize \textbf{Alphabet}: $\Sigma = \{\sigma_{1}, \ldots, \sigma_{M}\}$
    \item \scriptsize \textbf{Taille} de $\Sigma = \lvert \Sigma \rvert = M$
\end{itemize}

\begin{flushleft}
    \scriptsize \textbf{Fonction de codage}
\end{flushleft}

\begin{itemize}[leftmargin=0.5em,label=$\bullet$]
    \item \scriptsize $F : \Sigma \rightarrow T$
    \begin{itemize}[leftmargin=0.5em,label={}]
        \item \scriptsize  $\tau = F(\sigma)$
    \end{itemize}
\end{itemize}

\begin{flushleft}
    \scriptsize \textbf{Fonction de décodage}
\end{flushleft}


\begin{itemize}[leftmargin=0.5em,label={}]
    \item \scriptsize $F^-1 : T \rightarrow \Sigma$
    \begin{itemize}[label={}]
        \item \scriptsize $\sigma^\prime = F^-1(\tau^\prime)$
    \end{itemize}
    \item \scriptsize \textbf{si} $\tau^\prime \neq \tau \rightarrow \sigma^\prime \neq \sigma$
    \begin{itemize}[label=$\bullet$]
        \item \scriptsize erreur dans le canal
    \end{itemize}
    \item \scriptsize \textbf{si} $\tau^\prime = \tau \rightarrow \sigma^\prime = \sigma$
    \begin{itemize}[label=$\bullet$]
        \item \scriptsize sans erreur
    \end{itemize}
\end{itemize}

\footnotesize \subsection{Correction d'erreur}

\begin{flushleft}
    \scriptsize source $\rightarrow$ codeur $\rightarrow$ ajout syndrome $\Rightarrow$ \text{corr.\ erreur} $\rightarrow$ décodeur $\rightarrow$ récepteur
\end{flushleft}

\footnotesize \subsection{Entropie de Shannon}

\begin{itemize}[leftmargin=0.0em,label={}]
    \item \scriptsize $H_{b}(X) = \sum_{i=1}^{n} \log_{b}(1/P_{i}) = -\sum_{i=1}^{n} P_{i} \log_{b}(P_{i})$
    \begin{itemize}[label=$\bullet$]
        \item \scriptsize bit/symbole b=2
    \end{itemize}
    \item \scriptsize Si alphabet équiprobable :
    \begin{itemize}[label={}]
        \item $H_{2}(X) = \log_{2}(\lvert \Sigma \rvert)$
    \end{itemize}
    \item \scriptsize Si non équiprobable (fréquence des symboles) :
    \begin{itemize}[label={}]
        \item \scriptsize $\hat{H} = -\sum_{i=1}^{n} f_{i} \log_{2}(f_{i})$
    \end{itemize}
    \item \scriptsize  Entropie en bloc (exemple \textbf{$S^{3}$}) :
    \begin{itemize}[label=$\bullet$]
        \item \scriptsize $P(A) = P(B) = \ldots = P(Z) = 1/26$
        \item \scriptsize Probabilité des séquences - toutes équiprobable
        \item \scriptsize (AAA, AAB, AAC, \ldots, ZZZ), nb de séquences $= N = 26^{3} = 17576$
        \item \scriptsize Probabilité d'une séquence $S_{i}^{3} = 1/N = 1/17576$
        \item Donc,
    \end{itemize}
    \item \scriptsize $H = \log_{2}(N) = \log_{2}(N) = \log_{2}(26^{3})$
    \begin{itemize}[label={}]
        \item \scriptsize $H = \log_{2}(\lvert \Sigma \rvert^{x})$
        \item \scriptsize $H = x \cdot \log_{2}(\lvert \Sigma \rvert)$
        \item \scriptsize $H = 3 \times  \log_{2}(26)$
    \end{itemize}
    \item \scriptsize  Entropie par bloc
    \begin{itemize}[label={}]
        \item \scriptsize $\frac{H(S^{b})/b}{\log_{2}(N)}$
    \end{itemize}
\end{itemize}

\footnotesize \subsection{Modèle de Shannon révisé}

\begin{flushleft}
    \scriptsize source $\xrightarrow{\sigma}$ $\underset{F}{\text{codeur}}$ $\xrightarrow{\tau}$ $\underset{E}{\text{chiffr.}}$ $\xRightarrow{\tau^{\prime}}$ $\underset{D}{\text{déchiff.}}$ $\xrightarrow{\tau}$ $\underset{F^{-1}}{\text{décodeur}}$ $\xrightarrow{\sigma}$ récepteur
\end{flushleft}

\begin{flushleft}

\footnotesize \subsection{RSA: Abélien}

\begin{itemize}[label={}]
    \item \scriptsize $n = p \times q$
    \item \scriptsize $p,q$ sont des nombres premiers
    \item \scriptsize $\phi(n) = (p-1) \times (q-1)$
    \item \scriptsize clé publique $e$ $(n, e)$
    \begin{itemize}[label={}]
        \item \scriptsize $\text{pgcd}(e, \phi(n)) = 1$
        \item \scriptsize $1 < e < \phi(n)$
    \end{itemize}
    \item \scriptsize clé privé $d$ $(n, d)$
    \begin{itemize}[label={}]
        \item \scriptsize inverse de $e \mod(\phi(n))$
        \item \scriptsize $\frac{1 + k \times \phi(n)}{e}$
    \end{itemize}
    \item \scriptsize garantit que
    \item \scriptsize ${(M^{e})}^{d} \equiv M (\mod n)$
\end{itemize}

    \item \scriptsize clé privée: $x$

\footnotesize \section{Contrôle d'accès}

\begin{flushleft}
    \scriptsize \textbf{4 aspects}
\end{flushleft}

\begin{itemize}[label={}]
    \item \scriptsize Identification: Déterminer l'identité du demandeur
    \item \scriptsize Validation de l'identité du demandeur
    \item \scriptsize Validation du droit d'accès au ressources
    \item \scriptsize Audit(« Accounting »): Attribution d'actions à une identité
    \item \scriptsize $\rightarrow$ AAA(Authentification, Authorization and Accounting) ou IAAA
\end{itemize}

\begin{flushleft}
    \scriptsize \textbf{Dans un système d'exploitation}
\end{flushleft}

\begin{itemize}[label={}]
    \item \scriptsize Identification: nom usager, identificatuer de processus (PID)
    \item \scriptsize Authentification: commande d'authentification
    \item \scriptsize Autorisation: matrice d'accès, contrôleur de référence
    \item \scriptsize Audit: journaux (« logs »)
\end{itemize}

\footnotesize \subsection{Contrôleur de référence}

\begin{itemize}[label={}]
    \item \scriptsize Implémenter dans le noyau du système d'exploitation
    \item \scriptsize Maintient la sécurité au niveau voulu
    \item \scriptsize Vérifie chaque demande d'entrée selon une procédure stricte
    \item \scriptsize Composant qui s'interpose entre tous les accès de sujets à objets
    \item \scriptsize Sujet = Utilisateur ou processus
    \item \scriptsize Objet = Processus ou ressource (fichier)
    \item \scriptsize Mode d'accès = {R-Read, W-Write, X-Execute}
    \item \scriptsize Input: Requête d'accès (sujet, objet, mode d'accès)
    \item \scriptsize Output: Réponse (oui ou non) selon l'accès est permis ou pas
\end{itemize}

\footnotesize \subsection{Sous LINUX}

\begin{itemize}[label={}]
    \item \scriptsize Toutes les resources sont des objets (fichier, répertoire, mémoire, IO)
    \item \scriptsize Chaque objet a un propriétaire
    \item \scriptsize L'administrateur peut ajouter de nouveaux utilisateurs, lire et changer tous les objets, et changer les droits d'accès de tous les objets.
    \item \scriptsize Les utilisateurs peuvent seulement accéder aux objets pour lesquels ils ont la permission, et peuvent seulement changer les droits d'accès des objets dont ils sont propriétaires.
    \item \scriptsize Les logiciels s'exécutent avec les droits utilisateurs de l'utilisateur qui a lancé le programme
\end{itemize}
