\footnotesize \section{Modèle de Shannon}

\begin{center}
    \scriptsize source $\rightarrow$ codeur $\Rightarrow$ décodeur $\rightarrow$ récepteur
\end{center}

\begin{itemize}[leftmargin=0.5em,label=$\bullet$]
    \item \scriptsize \textbf{Alphabet}: $\Sigma = \{\sigma_{1}, \ldots, \sigma_{M}\}$
    \item \scriptsize \textbf{Taille} de $\Sigma = \lvert \Sigma \rvert = M$
\end{itemize}

\begin{flushleft}
    \scriptsize \textbf{Fonction de codage}
\end{flushleft}

\begin{itemize}[leftmargin=0.5em,label=$\bullet$]
    \item \scriptsize $F : \Sigma \rightarrow T$
    \begin{itemize}[leftmargin=0.5em,label={}]
        \item \scriptsize  $\tau = F(\sigma)$
    \end{itemize}
\end{itemize}

\begin{flushleft}
    \scriptsize \textbf{Fonction de décodage}
\end{flushleft}


\begin{itemize}[leftmargin=0.5em,label={}]
    \item \scriptsize $F^-1 : T \rightarrow \Sigma$
    \begin{itemize}[label={}]
        \item \scriptsize $\sigma^\prime = F^-1(\tau^\prime)$
    \end{itemize}
    \item \scriptsize \textbf{si} $\tau^\prime \neq \tau \rightarrow \sigma^\prime \neq \sigma$
    \begin{itemize}[label=$\bullet$]
        \item \scriptsize erreur dans le canal
    \end{itemize}
    \item \scriptsize \textbf{si} $\tau^\prime = \tau \rightarrow \sigma^\prime = \sigma$
    \begin{itemize}[label=$\bullet$]
        \item \scriptsize sans erreur
    \end{itemize}
\end{itemize}

\footnotesize \subsection{Correction d'erreur}

\begin{flushleft}
    \scriptsize source $\rightarrow$ codeur $\rightarrow$ ajout syndrome $\Rightarrow$ \text{corr.\ erreur} $\rightarrow$ décodeur $\rightarrow$ récepteur
\end{flushleft}

\footnotesize \subsection{Entropie de Shannon}

\begin{itemize}[leftmargin=0.0em,label={}]
    \item \scriptsize $H_{b}(X) = \sum_{i=1}^{n} \log_{b}(1/P_{i}) = -\sum_{i=1}^{n} P_{i} \log_{b}(P_{i})$
    \begin{itemize}[label=$\bullet$]
        \item \scriptsize bit/symbole b=2
    \end{itemize}
    \item \scriptsize Si alphabet équiprobable :
    \begin{itemize}[label={}]
        \item $H_{2}(X) = \log_{2}(\lvert \Sigma \rvert)$
    \end{itemize}
    \item \scriptsize Si non équiprobable (fréquence des symboles) :
    \begin{itemize}[label={}]
        \item \scriptsize $\hat{H} = -\sum_{i=1}^{n} f_{i} \log_{2}(f_{i})$
    \end{itemize}
    \item \scriptsize  Entropie en bloc (exemple \textbf{$S^{3}$}) :
    \begin{itemize}[label=$\bullet$]
        \item \scriptsize $P(A) = P(B) = \ldots = P(Z) = 1/26$
        \item \scriptsize Probabilité des séquences - toutes équiprobable
        \item \scriptsize (AAA, AAB, AAC, \ldots, ZZZ), nb de séquences $= N = 26^{3} = 17576$
        \item \scriptsize Probabilité d'une séquence $S_{i}^{3} = 1/N = 1/17576$
        \item Donc,
    \end{itemize}
    \item \scriptsize $H = \log_{2}(N) = \log_{2}(N) = \log_{2}(26^{3})$
    \begin{itemize}[label={}]
        \item \scriptsize $H = \log_{2}(\lvert \Sigma \rvert^{x})$
        \item \scriptsize $H = x \cdot \log_{2}(\lvert \Sigma \rvert)$
        \item \scriptsize $H = 3 \times  \log_{2}(26)$
    \end{itemize}
    \item \scriptsize  Entropie par bloc
    \begin{itemize}[label={}]
        \item \scriptsize $\frac{H(S^{b})/b}{\log_{2}(N)}$
    \end{itemize}
\end{itemize}

\footnotesize \subsection{Modèle de Shannon révisé}

\begin{flushleft}
    \scriptsize source $\xrightarrow{\sigma}$ $\underset{F}{\text{codeur}}$ $\xrightarrow{\tau}$ $\underset{E}{\text{chiffr.}}$ $\xRightarrow{\tau^{\prime}}$ $\underset{D}{\text{déchiff.}}$ $\xrightarrow{\tau}$ $\underset{F^{-1}}{\text{décodeur}}$ $\xrightarrow{\sigma}$ récepteur
\end{flushleft}

\begin{flushleft}

\footnotesize \subsection{RSA: Abélien}

\begin{itemize}[label={}]
    \item \scriptsize $n = p \times q$
    \item \scriptsize $p,q$ sont des nombres premiers
    \item \scriptsize $\phi(n) = (p-1) \times (q-1)$
    \item \scriptsize clé publique $e$ $(n, e)$
    \begin{itemize}[label={}]
        \item \scriptsize $\text{pgcd}(e, \phi(n)) = 1$
        \item \scriptsize $1 < e < \phi(n)$
    \end{itemize}
    \item \scriptsize clé privé $d$ $(n, d)$
    \begin{itemize}[label={}]
        \item \scriptsize inverse de $e \mod(\phi(n))$
        \item \scriptsize $\frac{1 + k \times \phi(n)}{e}$
    \end{itemize}
    \item \scriptsize garantit que
    \item \scriptsize ${(M^{e})}^{d} \equiv M (\mod n)$
\end{itemize}

\footnotesize \section{Contrôle d'accès}

\begin{flushleft}
    \scriptsize \textbf{4 aspects}
\end{flushleft}

\begin{itemize}[label={}]
    \item \scriptsize Identification: Déterminer l'identité du demandeur
    \item \scriptsize Validation de l'identité du demandeur
    \item \scriptsize Validation du droit d'accès au ressources
    \item \scriptsize Audit(« Accounting »): Attribution d'actions à une identité
    \item \scriptsize $\rightarrow$ AAA(Authentification, Authorization and Accounting) ou IAAA
\end{itemize}

\begin{flushleft}
    \scriptsize \textbf{Dans un système d'exploitation}
\end{flushleft}

\begin{itemize}[label={}]
    \item \scriptsize Identification: nom usager, identificatuer de processus (PID)
    \item \scriptsize Authentification: commande d'authentification
    \item \scriptsize Autorisation: matrice d'accès, contrôleur de référence
    \item \scriptsize Audit: journaux (« logs »)
\end{itemize}

\footnotesize \subsection{Contrôleur de référence}

\begin{itemize}[label={}]
    \item \scriptsize Implémenter dans le noyau du système d'exploitation
    \item \scriptsize Maintient la sécurité au niveau voulu
    \item \scriptsize Vérifie chaque demande d'entrée selon une procédure stricte
    \item \scriptsize Composant qui s'interpose entre tous les accès de sujets à objets
    \item \scriptsize Sujet = Utilisateur ou processus
    \item \scriptsize Objet = Processus ou ressource (fichier)
    \item \scriptsize Mode d'accès = {R-Read, W-Write, X-Execute}
    \item \scriptsize Input: Requête d'accès (sujet, objet, mode d'accès)
    \item \scriptsize Output: Réponse (oui ou non) selon l'accès est permis ou pas
\end{itemize}

\footnotesize \subsection{Sous LINUX}

\begin{itemize}[label={}]
    \item \scriptsize Toutes les resources sont des objets (fichier, répertoire, mémoire, IO)
    \item \scriptsize Chaque objet a un propriétaire
    \item \scriptsize L'administrateur peut ajouter de nouveaux utilisateurs, lire et changer tous les objets, et changer les droits d'accès de tous les objets.
    \item \scriptsize Les utilisateurs peuvent seulement accéder aux objets pour lesquels ils ont la permission, et peuvent seulement changer les droits d'accès des objets dont ils sont propriétaires.
    \item \scriptsize Les logiciels s'exécutent avec les droits utilisateurs de l'utilisateur qui a lancé le programme
\end{itemize}

    \item \scriptsize Group ID (GID)
    \item \scriptsize Les utilisateur ont un groupe principal
    \item \scriptsize Les utilisateurs peuvent rejoindre d'autres groupes
    \item \scriptsize Les fichiers ont le groupe principal de l'utilisateur qui a créé le fihcier
\end{itemize}

\begin{flushleft}
    \scriptsize Utilisateurs et groupes ne sont pas des objets
\end{flushleft}

\begin{itemize}[label=$\bullet$]
    \item \scriptsize Chaque fichier a un UID et GID assigné
    \item \scriptsize Chaque programme a un UID et GID assigné
    \item \scriptsize Avant d'exécuter un appel de fonction du système (« system call »), le contrôleur de référence vérifie:
    \begin{itemize}[label=$\circ$]
        \item \scriptsize Si UID = 0, permettre l'accès
        \item \scriptsize Sinon, lire la liste de contrôle d'accès de l'objet et vérifier si l'accès est permis
    \end{itemize}
\end{itemize}

\footnotesize \subsection{DAC et MAC}

\begin{itemize}[label=$\bullet$]
    \item \scriptsize Linux utilise le \textbf{Contrôle d'accès discrétionnaire (DAC)}
    \begin{itemize}[label=$\circ$]
        \item \scriptsize Les utilisateurs peuvent changer les permissions de leur fichiers
    \end{itemize}
    \item \scriptsize DAC représente les droits sous forme d'une matrice
    \item \scriptsize DAC fonctionne correctement sous 2 conditions
    \begin{itemize}[label=$\rightarrow$]
        \item \scriptsize Si les usagers ne font pas d'erreurs
        \item \scriptsize On peut faire confiance à tous les programmes
    \end{itemize}
\end{itemize}

\begin{center}
    \includegraphics[width=1\linewidth]{static/limites-dac.png}
\end{center}

\begin{itemize}[label=$\bullet$]
    \item \scriptsize MAC = Mandatory Access Control
    \begin{itemize}[label=$\circ$]
        \item \scriptsize Contrôle d'accès obligatoire
    \end{itemize}
    \item scriptsize MAC ajoute des étiquettes à chaque sujet et objet
    \item \scriptsize Une politique d'accès contient les règles d'accès
    \item \scriptsize Politique par défaut
    \begin{itemize}[label=$\circ$]
        \item \scriptsize REFUSÉ (« Deny »): l'accès n'est pas permis, à moins qu'il y ait une règle indiquant le contraire dans la politique d'accès
    \end{itemize}
    \item \scriptsize Les règles et étiquettes peuvent seulement être changées par un administrateur avec un logiciel de confiance (« Trusted »)
    \item \scriptsize Exemple d'étiquettes (niveau de sécurité):
    \begin{itemize}[label={}]
        \item \scriptsize Public $\le$ Confidentiel $\le$ Secret
    \end{itemize}
\end{itemize}

\begin{flushleft}
    \scriptsize  Condition de sécurité (Modèle de Bell et LaPadula)
\end{flushleft}

\begin{itemize}[label=$\bullet$]
    \item \scriptsize  No Read Up
    \begin{itemize}[label=$\circ$]
        \item \scriptsize Un sujet s peut lire un objet o si:
        \begin{itemize}[label={}]
            \item \scriptsize $\text{niveau\_classification(o)} \le \text{niveau\_habilitation(s)}$
        \end{itemize}
    \end{itemize}
    \item \scriptsize  No Write Down
    \begin{itemize}[label=$\circ$]
        \item \scriptsize Un sujet s peut modifier un objet o si:
        \begin{itemize}[label={}]
            \item \scriptsize $\text{niveau\_habilitation(s)} \le \text{niveau\_classification(o)}$
        \end{itemize}
    \end{itemize}
    \item Objectif du « No write down »
    \begin{itemize}[label=$\circ$]
        \item \scriptsize Soit un programme s'exécutant au niveau « Secret »
        \item \scriptsize Ce programme peut lire des données classées « Secret »
        \item \scriptsize Mais le « No write down » empêche un éventuel piège contenu dans ce programme de transmettre les données lues vers un utilisateur qui ne serait pas habilité au niveau « Secret »
    \end{itemize}
\end{itemize}

\footnotesize \subsection{RBAC}

\begin{itemize}[label=$\bullet$]
    \item \scriptsize RBAC = Role Based Access Control
    \item \scriptsize RBAC est basé sur deux points
    \begin{itemize}[label=$\circ$]
        \item \scriptsize Dans les organisations les employés sont affectés à des rôles
        \begin{itemize}[label={}]
            \item \scriptsize Comptable, programmeur, docteur, infirmière, technicien, $\ldots$
            \item \scriptsize Les rôles sont organisé en hiéarchies
        \end{itemize}
        \item le fait que chaque employé, pour exécuter son rôle, a besoin de certaines permissions
    \end{itemize}
    \item \scriptsize RBAC s'appuie sur la notion organisationnelle de rôle pour associer des permissions de sécurité aux différents rôles
    \item \scriptsize Le rôle devient un mécanisme pour associer des permissions aux usagers
\end{itemize}

\begin{center}
    \includegraphics[width=0.5\linewidth]{static/rbac.png}
\end{center}

\begin{itemize}[label=$\bullet$]
    \item \scriptsize Pour utiliser ses rôles, un usager doit activer des sessions
    \item \scriptsize Une session est un processus qui agit pour un usager
    \item \scriptsize Pour accéder à une session, un sujet doit s'authentifier
    \item \scriptsize Un sujet peut se trouver dans plusieurs sessions simultanément
\end{itemize}

\begin{center}
    \includegraphics[width=1\linewidth]{static/rbac-session.png}
\end{center}

\begin{itemize}[label=$\bullet$]
    \item \scriptsize On peut introduire une hiéarchie de rôles
    \item \scriptsize Héritage des permissions
\end{itemize}

\begin{itemize}[label=$\bullet$]
    \item \scriptsize  Séparation des pouvoirs
    \begin{itemize}[label=$\circ$]
        \item \scriptsize Ce sont les contraintes les plus importantes
        \item \scriptsize Exemple: Celui qui approuve un chèque (rôle R1) ne peut pas être celui qui le signe (rôle R2)
        \item \scriptsize Dans RBAC, cela se traduit par une contraintes qui rend impossible qu'un utilisateur soit affectés à R1 et R2
    \end{itemize}
    \item \scriptsize SOD statique (SSOD) ou dynamique (DSOD)
    \begin{itemize}[label=$\circ$]
        \item \scriptsize SSOD: la séparation s'applique en toute situation
        \item \scriptsize DSOD: la séparation s'applique uniquement dans une même session
    \end{itemize}
\end{itemize}

\begin{itemize}[label=$\bullet$]
    \item \scriptsize AGLP
    \begin{itemize}[label=$\circ$]
        \item \scriptsize Access - Global - Local - Permission
        \item \scriptsize Implémentation de RBAC sous Windows
        \item \scriptsize Repose sur les Active Directory(AD)
    \end{itemize}
    \item \scriptsize Éléments
    \begin{itemize}[label=$\circ$]
        \item \scriptsize Groupe « Globaux »
        \begin{itemize}[label=$\bullet$]
            \item \scriptsize regroupement des utilisateurs, typiquement selon leur rôles
            \item \scriptsize généralement définis sur des serveurs de domaine globaux (d'où le nom)
        \end{itemize}
        \item \scriptsize Groupe « Locaux »
        \begin{itemize}[label=$\bullet$]
            \item \scriptsize auxquels sont attribués des Permissions sur des ressources
            \item \scriptsize généralement définis et résidans sur les serveurs où ces ressources se trouvent (d'où le nom)
            \item \scriptsize les membres sont exclusivement des groupes globaux
        \end{itemize}
    \end{itemize}
    \item \scriptsize Principes de base
    \begin{itemize}[label=$\circ$]
        \item \scriptsize On n'attribue pas de permissions aux groupes globaux ni aux utilisateurs
        \item \scriptsize On n'ajoute pas d'utilisateurs dans les groupes locaux
        \item \scriptsize Séparation des responsabilités
        \begin{itemize}[label={}]
            \item \scriptsize Gestion des usager: U et G
            \item \scriptsize Politique de sécurité: G et L
            \item \scriptsize Administrateur de système: L et R
        \end{itemize}
    \end{itemize}
\end{itemize}

\footnotesize \subsection{Pare-feu}

\begin{itemize}[label=$\bullet$]
    \item \scriptsize Le pare-feu est un équipement ou logiciel qui agit en tant que filtre réseau
    \item \scriptsize Un pare-feu doit être configuré conformément à une politique de sécurité qui définit les paquets autorisés à traverser le pare-feu (et aussi les interdictions) 
    \item \scriptsize Politique de sécurité = Ensemble de règles (Règle = ACL (Access Control List))
    \item \scriptsize Équipement en coupure entre 2 ou plusieurs réseaux
    \begin{itemize}[label=$\circ$]
        \item \scriptsize inspecte les paquets réseaux traversant le pare-feu
        \item \scriptsize Le pare-feu ne transmet que les paquets qui respectent les règles de filtrage implémentées dans la configuration du pare-feu
    \end{itemize}
\end{itemize}

\begin{flushleft}
    \scriptsize \textbf{Pare-feu statique}
\end{flushleft}

\begin{itemize}[label=$\bullet$]
    \item \scriptsize Examine paquet par paquet
    \item \scriptsize Conserve pas d'informationm sur l'état
    \item \scriptsize Le pare-feu inspecte les paquets réseau en se basant sur les informations de l'en-tête du paquet
    \item \scriptsize Le pare-feu prend une décision (pass, block, log) en fonction des règles de la politique de sécurité
    \item \scriptsize La décision de filtrage dépend uniquement des données des couches 2,3 et 4
    \begin{itemize}[label=$\circ$]
        \item \scriptsize Addresse IP source/destination
        \item \scriptsize Port source/destination
        \item \scriptsize Type de protocole utilisé (TCP/UDP/ICMP)
        \item \scriptsize Signalisation du paquet (SYN, ACK)
        \item \scriptsize Addresses MAC
    \end{itemize}
    \item \scriptsize Peut être implémenté en logiciel (ex: IPtables) ou matériel (ex: routeur Cisco)
\end{itemize}

\begin{flushleft}
    \scriptsize \textbf{Pare-feu dynamique}
\end{flushleft}

\begin{itemize}[label=$\bullet$]
    \item \scriptsize Examine paquet par paquet, mais essaie d'établir des relations entre paquets
    \begin{itemize}[label=$\circ$]
        \item \scriptsize UDP
        \begin{itemize}[label={}]
            \item \scriptsize Associe le paquet avec d'autres paquets sur les mêmes ports et addresses
            \item \scriptsize En général, permet seulement des réponses si requêtes originales venant d'addresses internes
        \end{itemize}
        \item \scriptsize TCP
        \begin{itemize}[label={}]
            \item \scriptsize Garde l'information sur l'état et direction de la session TCP
            \item \scriptsize Regarde en plus les flags TCP pour déterminer si hors-protocole
        \end{itemize}
        \item \scriptsize Applications qui changent de port (e.g. FTP)
        \begin{itemize}[label={}]
            \item \scriptsize Suit et autorise les port éphémères utilisés par les applications
        \end{itemize}
    \end{itemize}
    \item \scriptsize Ces pare-feu sont limités aux couches 2 à 4 du modèle OSI
    \begin{itemize}[label=$\circ$]
        \item \scriptsize Pas de connaissance de l'état de la connexion au niveau applicatifs
        \item \scriptsize Usurpation de port
    \end{itemize}
    \item \scriptsize Plusieurs attaques passent par les ports ouvert
    \item \begin{itemize}[label=$\circ$]
        \item \scriptsize Injection SQL
        \item \scriptsize Attaques de force brute
    \end{itemize}
    \item \scriptsize En augmentant l'intelligence d'un pare-feu pour interpréter les protocoles applicatifs ont peut obtenir une meilleur analyse du trafic
    \item \begin{itemize}[label=$\circ$]
        \item \scriptsize Pare-feu applicatif
        \item \scriptsize Serveur mandataire ou « proxy »
    \end{itemize}
\end{itemize}
